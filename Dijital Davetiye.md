        // --- 1. GERİ SAYIM SAYACI SİSTEMİ ---
        // Hedef düğün tarihi (Yıl, Ay-1, Gün, Saat, Dakika, Saniye)
        // Not: Javascript'te Aylar 0'dan başlar (Ocak=0, Ağustos=7).
        const weddingDate = new Date(2026, 7, 22, 19, 0, 0).getTime();

        const countdownInterval = setInterval(function() {
            const now = new Date().getTime();
            const difference = weddingDate - now;

            // Zaman birimlerinin matematiksel hesaplamaları
            const days = Math.floor(difference / (1000 * 60 * 60 * 24));
            const hours = Math.floor((difference % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutes = Math.floor((difference % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((difference % (1000 * 60)) / 1000);

            // DOM elemanlarını güncelleme (Sıfır ekleme formatı ile)
            document.getElementById("days").innerText = days < 10 ? "0" + days : days;
            document.getElementById("hours").innerText = hours < 10 ? "0" + hours : hours;
            document.getElementById("minutes").innerText = minutes < 10 ? "0" + minutes : minutes;
            document.getElementById("seconds").innerText = seconds < 10 ? "0" + seconds : seconds;

            // Tarih geçtiyse sayacı sıfırla veya durdur
            if (difference < 0) {
                clearInterval(countdownInterval);
                document.getElementById("wedding-countdown").innerHTML = "<p style='font-family: var(--font-serif); font-size: 1.5rem; color: var(--accent-color);'>Biz Evlendik!</p>";
            }
        }, 1000);


        // --- 2. FORMA ÖZEL DİNAMİK ALAN (+1 KONTROLÜ) ---
        const attendanceOptions = document.querySelectorAll('input[name="attendance"]');
        const guestSection = document.getElementById('guestSection');
        const bringingGuestCheckbox = document.getElementById('bringingGuest');

        attendanceOptions.forEach(option => {
            option.addEventListener('change', function() {
                if (this.value === 'will-attend') {
                    // Sadece katılacağım seçildiğinde +1 seçeneği görünür olur
                    guestSection.style.display = 'block';
                } else {
                    // Diğer durumlarda gizlenir ve kutu işaretliyse sıfırlanır
                    guestSection.style.display = 'none';
                    bringingGuestCheckbox.checked = false;
                }
            });
        });


        // --- 3. RSVP FORM SUBMIT (VERİ YÖNETİMİ) ---
        document.getElementById('wedding-rsvp-form').addEventListener('submit', function(e) {
            e.preventDefault(); // Sayfa yenilenmesini engelle

            // Form datalarını değişkenlere toplama
            const fullName = document.getElementById('fullName').value;
            const email = document.getElementById('email').value;
            const attendance = document.querySelector('input[name="attendance"]:checked').value;
            const isBringingGuest = bringingGuestCheckbox.checked;
            const userMessage = document.getElementById('message').value;

            // Antigravity veya backend entegrasyonu için obje yapısı
            const rsvpData = {
                name_surname: fullName,
                email_address: email,
                status: attendance,
                plus_one: isBringingGuest,
                message: userMessage,
                timestamp: new Date().toISOString()
            };

            // Konsol çıktısı (Verinin kontrolü için)
            console.log("RSVP Gönderilen Veri:", rsvpData);

            // Kullanıcıya şık bir geri bildirim verme
            alert("Yanıtınız başarıyla kaydedildi. Teşekkür ederiz!");
            
            // Formu sıfırlama
            this.reset();
            guestSection.style.display = 'none';
        });
    </script>
</body>
